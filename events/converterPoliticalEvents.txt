country_event = {
	id = conv.political.1
	title = conv.political.1.t
	desc = conv.political.1.d
	picture = GFX_report_event_generic_factory


	trigger = {
		NOT = { has_government = absolutist }
		absolutist > 0.6
		NOT = { absolutist > 0.7 }
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 30
	}

	option = {
		name = conv.political.1.a
		add_political_power = -100
		set_politics = {
			ruling_party = absolutist
			elections_allowed = no
		}
	}

	option = {
		name = conv.political.1.b
		ai_chance = {
			factor = 0
		}
		custom_effect_tooltip = impending_civil_war_tt
	}
}

country_event = {
	id = conv.political.9
	title = conv.political.9.t
	desc = conv.political.9.d
	picture = GFX_report_event_generic_factory


	trigger = {
		absolutist > 0.7
		num_of_controlled_states > 1
		NOT = { has_government = absolutist }
		has_civil_war = no
	}

	mean_time_to_happen = {
		days = 2
	}

	immediate = {
		hidden_effect = { remove_ideas = absolutist_revolutionaries }
	}

	option = {
		name = conv.political.9.a
		ai_chance = {
			factor = 0
		}
		start_civil_war = {
			ideology = absolutist
			size = 0.5
		}
	}

	option = {
		name = conv.political.9.b
		ai_chance = {
			factor = 0
		}
		if = {
			limit = {
				has_government = fascism
			}
			start_civil_war = {
				ruling_party = absolutist
				ideology = fascism
				size = 0.5
			}
		}
		if = {
			limit = {
				has_government = neutrality
			}
			start_civil_war = {
				ruling_party = absolutist
				ideology = neutrality
				size = 0.5
			}
		}
	}

	option = {
		name = conv.political.9.c
		trigger = {
			is_ai = yes
		}
		add_political_power = -100
		set_politics = {
			ruling_party = absolutist
			elections_allowed = no
		}
	}
}
